#include <stdio.h>
#include <stdlib.h>
char* f(a)
{
    int i;
    if ( a == 1 )
    {
        return "Непростое";
    }
    if ( a == 0)
    {
        return "Непростое";
    }
    for (i = 2; i <= a/2; i+=1 )
    {
        if ((a/i)*i == a)
        {
            return "Непростое";
        }
    return "Простое";
    }
}
int main()
{
    int a;
    scanf("%d", &a);
    printf("%s", f(a));
    return 0;
}
