#include <stdio.h>
#include <stdlib.h>
int C(n,k)
{
    if (n == k)
    {
        return 1;
    }
    if (k == 0)
    {
        return 1;
    }
    return C(n-1, k) + C(n-1, k-1);
}




int main()
{
    int n, i;
    scanf("%d", &n);
    int mas[999999];
    for (i = 0; i<=n; i+=1)
    {
        mas[i] = C(n, i);
    }
    for (i = 0; i<=n; i+=1)
    {
        printf("%d ", mas[i]);

    }


    return 0;
}
